Dependências:

Uso:

def make_serializable(val):
    """Converte qualquer valor EXIF para tipo serializável em JSON."""
    if isinstance(val, IFDRational):
        return float(val)
    if isinstance(val, bytes):
        decoded_ascii = None
        decoded_utf16 = None
        try:
            s = val.decode("ascii").rstrip("\x00")
            if s and s.isprintable():
                decoded_ascii = s
        except Exception:
            pass
        try:
            s = val.decode("utf-16-le").rstrip("\x00")
            if s and s.isprintable():
                decoded_utf16 = s
        except Exception:
            pass
        return {
            "_type": "bytes",
            "hex": val.hex(),
            "length_bytes": len(val),
            "decoded_ascii": decoded_ascii,
            "decoded_utf16le": decoded_utf16,
        }
    if isinstance(val, (tuple, list)):
        return [make_serializable(v) for v in val]
    if isinstance(val, dict):
        return {k: make_serializable(v) for k, v in val.items()}
    return val


def dms_to_decimal(dms, ref):
    """Converte DMS (graus, minutos, segundos) + referência para decimal com sinal."""
    deg, mn, sec = (float(v) for v in dms)
    decimal = deg + mn / 60.0 + sec / 3600.0
    if ref in ("S", "W"):
        decimal = -decimal
    return round(decimal, 10)

def extract_exif(img):
    """Extrai todos os campos EXIF incluindo GPS e MakerNote."""
    raw = img._getexif()
    if not raw:
        return {}

    result = {}
    for tag_id, value in raw.items():
        tag = TAGS.get(tag_id, str(tag_id))
        if tag == "GPSInfo":
            gps = {}
            for k, v in value.items():
                gps_tag = GPSTAGS.get(k, str(k))
                gps[gps_tag] = make_serializable(v)
            result["GPSInfo"] = gps
        else:
            result[tag] = make_serializable(value)
    return result


def parse_gps_section(exif_raw):
    """Extrai e interpreta coordenadas GPS do bloco EXIF."""
    gps = exif_raw.get("GPSInfo", {})
    if not gps:
        return {}

    lat_dms = gps.get("GPSLatitude")
    lon_dms = gps.get("GPSLongitude")
    lat_ref = gps.get("GPSLatitudeRef", "N")
    lon_ref = gps.get("GPSLongitudeRef", "E")

    lat_dec = dms_to_decimal(lat_dms, lat_ref) if lat_dms else None
    lon_dec = dms_to_decimal(lon_dms, lon_ref) if lon_dms else None

    alt_raw = gps.get("GPSAltitude")
    alt = float(alt_raw) if alt_raw is not None else None

    alt_ref_raw = gps.get("GPSAltitudeRef")
    if isinstance(alt_ref_raw, dict):
        alt_ref_byte = alt_ref_raw.get("raw_list", [0])[0]
    else:
        alt_ref_byte = 0

    return {
        "datum": gps.get("GPSMapDatum", "WGS-84").strip("\x00"),
        "gps_version_id": gps.get("GPSVersionID", {}).get("raw_list"),
        "gps_status_exif": gps.get("GPSStatus"),
        "latitude_dms": lat_dms,
        "longitude_dms": lon_dms,
        "latitude_ref": lat_ref,
        "longitude_ref": lon_ref,
        "latitude_decimal_exif": lat_dec,
        "longitude_decimal_exif": lon_dec,
        "altitude_m_exif": alt,
        "altitude_ref_byte": alt_ref_byte,
        "altitude_ref_description": "Above sea level" if alt_ref_byte == 0 else "Below sea level",
        "google_maps_url": (
            f"https://maps.google.com/?q={lat_dec},{lon_dec}"
            if lat_dec and lon_dec else None
        ),
    }

def extract_xmp(data: bytes):
    """Localiza e parseia o bloco XMP/RDF embutido no JPEG."""
    start = data.find(b"<x:xmpmeta")
    end = data.find(b"</x:xmpmeta>")
    if start == -1 or end == -1:
        return None, {}

    xmp_raw = data[start : end + len(b"</x:xmpmeta>")].decode("utf-8", errors="replace")

    # Parse de atributos via busca de padrão  key="value"
    import re
    attrs = {}
    for m in re.finditer(r'([\w:.\-]+)="([^"]*)"', xmp_raw):
        key = m.group(1)
        val = m.group(2)
        # Tenta converter números
        try:
            if "." in val or val.lstrip("+-").replace(".", "").isdigit():
                val_num = float(val)
                attrs[key] = val_num if "." in val else int(val_num)
            elif val.lstrip("+-").isdigit():
                attrs[key] = int(val)
            else:
                attrs[key] = val
        except (ValueError, AttributeError):
            attrs[key] = val

    return xmp_raw, attrs


def parse_dji_xmp(attrs: dict):
    """Organiza os atributos XMP DJI em seções estruturadas."""

    def get(key, default=None):
        return attrs.get(f"drone-dji:{key}", attrs.get(key, default))

    def getf(key, default=None):
        v = get(key, default)
        try:
            return float(v) if v is not None else default
        except (ValueError, TypeError):
            return default

    def geti(key, default=None):
        v = get(key, default)
        try:
            return int(v) if v is not None else default
        except (ValueError, TypeError):
            return default

    # --- Dewarp parsing ---
    def parse_dewarp(raw, model="K3"):
        if not raw:
            return None
        try:
            date_str, vals_str = raw.split(";")
            vals = [float(v) for v in vals_str.split(",")]
            base = {
                "calibration_date": date_str,
                "fx_px": vals[0],
                "fy_px": vals[1],
                "cx_offset_px": vals[2],
                "cy_offset_px": vals[3],
                "k1": vals[4],
                "k2": vals[5],
                "p1": vals[6],
                "p2": vals[7],
            }
            if model == "K6" and len(vals) >= 12:
                base.update({"k3": vals[8], "k4": vals[9], "k5": vals[10], "k6": vals[11]})
            elif len(vals) >= 9:
                base["k3"] = vals[8]
            return base
        except Exception:
            return {"raw": raw}

    return {
        "dji_xmp_spec_version": get("Version"),
        "image_source": get("ImageSource"),
        "gps_status": get("GpsStatus"),
        "altitude_type": get("AltitudeType"),
        "gps_latitude": getf("GpsLatitude"),
        "gps_longitude": getf("GpsLongitude"),
        "absolute_altitude_m": getf("AbsoluteAltitude"),
        "relative_altitude_m": getf("RelativeAltitude"),
        "gimbal_roll_deg": getf("GimbalRollDegree"),
        "gimbal_yaw_deg": getf("GimbalYawDegree"),
        "gimbal_pitch_deg": getf("GimbalPitchDegree"),
        "flight_roll_deg": getf("FlightRollDegree"),
        "flight_yaw_deg": getf("FlightYawDegree"),
        "flight_pitch_deg": getf("FlightPitchDegree"),
        "flight_x_speed_ms": getf("FlightXSpeed"),
        "flight_y_speed_ms": getf("FlightYSpeed"),
        "flight_z_speed_ms": getf("FlightZSpeed"),
        "cam_reverse": geti("CamReverse"),
        "gimbal_reverse": geti("GimbalReverse"),
        "sensor_temperature_celsius": getf("SensorTemperature"),
        "product_name": get("ProductName"),
        "self_data": get("SelfData"),
        "rtk_flag": geti("RtkFlag"),
        "rtk_std_lon_m": getf("RtkStdLon"),
        "rtk_std_lat_m": getf("RtkStdLat"),
        "rtk_std_hgt_m": getf("RtkStdHgt"),
        "rtk_diff_age_sec": getf("RtkDiffAge"),
        "surveying_mode": geti("SurveyingMode"),
        "dewarp_flag": geti("DewarpFlag"),
        "dewarp_model_K3": parse_dewarp(get("DewarpData"), "K3"),
        "dewarp_model_K6": parse_dewarp(get("DewarpDataK6"), "K6"),
        "calibrated_focal_length_px": getf("CalibratedFocalLength"),
        "calibrated_optical_center_x_px": getf("CalibratedOpticalCenterX"),
        "calibrated_optical_center_y_px": getf("CalibratedOpticalCenterY"),
        "utc_at_exposure": get("UTCAtExposure"),
        "shutter_type": get("ShutterType"),
        "shutter_count": geti("ShutterCount"),
        "camera_serial_number": get("CameraSerialNumber"),
        "drone_model": get("DroneModel"),
        "drone_serial_number": get("DroneSerialNumber"),
        "lrf_status": get("LRFStatus"),
        "lrf_target_distance_m": getf("LRFTargetDistance"),

# ══════════════════════════════════════════════════════════════════════════════
#  SEGMENTOS JPEG (análise binária)
# ══════════════════════════════════════════════════════════════════════════════

MARKER_NAMES = {
    0xC0: "SOF0 (Baseline DCT)",
    0xC1: "SOF1 (Extended Sequential DCT)",
    0xC2: "SOF2 (Progressive DCT)",
    0xC4: "DHT (Huffman Table)",
    0xD8: "SOI (Start of Image)",
    0xD9: "EOI (End of Image)",
    0xDA: "SOS (Start of Scan)",
    0xDB: "DQT (Quantization Table)",
    0xDD: "DRI (Restart Interval)",
    0xE0: "APP0 (JFIF)",
    0xE1: "APP1 (EXIF/XMP)",
    0xE2: "APP2 (ICC/MPF)",
    0xE7: "APP7 (DJI proprietary)",
    0xE8: "APP8 (DJI proprietary)",
    0xFE: "COM (Comment)",
}


def scan_jpeg_segments(data: bytes):
    """Varre todos os marcadores JPEG e retorna lista de segmentos."""
    segments = []
    i = 0
    while i < len(data) - 1:
        if data[i] != 0xFF or data[i + 1] in (0x00, 0xFF):
            i += 1
            continue

        marker_byte = data[i + 1]
        offset = i

        if marker_byte in (0xD8, 0xD9):  # SOI / EOI — sem campo de comprimento
            segments.append({
                "offset_dec": offset,
                "offset_hex": f"0x{offset:08X}",
                "marker": f"FF{marker_byte:02X}",
                "name": MARKER_NAMES.get(marker_byte, f"MARKER_{marker_byte:02X}"),
                "length_bytes": None,
            })
            i += 2
        elif i + 4 <= len(data):
            length = struct.unpack(">H", data[i + 2 : i + 4])[0]
            name = MARKER_NAMES.get(
                marker_byte,
                f"APP{marker_byte - 0xE0}" if 0xE0 <= marker_byte <= 0xEF
                else f"MARKER_{marker_byte:02X}",
            )
            segments.append({
                "offset_dec": offset,
                "offset_hex": f"0x{offset:08X}",
                "marker": f"FF{marker_byte:02X}",
                "name": name,
                "length_bytes": length,
            })
            i += 2 + length
        else:
            i += 1

    return segments


# ══════════════════════════════════════════════════════════════════════════════
#  MAKERNOTE DJI (IFD proprietário little-endian)
# ══════════════════════════════════════════════════════════════════════════════

IFD_TYPES = {
    1: "BYTE", 2: "ASCII", 3: "SHORT", 4: "LONG", 5: "RATIONAL",
    7: "UNDEFINED", 9: "SLONG", 10: "SRATIONAL", 11: "FLOAT", 12: "DOUBLE",
}


def parse_makernote(raw_bytes: bytes):
    """Decodifica o MakerNote DJI como IFD little-endian."""
    if not raw_bytes or len(raw_bytes) < 4:
        return None

    # raw_bytes já é o valor bytes do campo MakerNote
    data = raw_bytes if isinstance(raw_bytes, bytes) else bytes(raw_bytes)

    try:
        num_entries = struct.unpack("<H", data[:2])[0]
        if num_entries > 50:  # sanity check
            return {"_note": "IFD entry count implausible, skipping parse", "raw_hex": data.hex()}

        entries = []
        for i in range(num_entries):
            off = 2 + i * 12
            if off + 12 > len(data):
                break
            tag, typ, count = struct.unpack("<HHI", data[off : off + 8])
            val_bytes = data[off + 8 : off + 12]

            interpreted = None
            if typ == 1 and count == 1:   # BYTE
                interpreted = val_bytes[0]
            elif typ == 3 and count == 1: # SHORT
                interpreted = struct.unpack("<H", val_bytes[:2])[0]
            elif typ == 4 and count == 1: # LONG
                interpreted = struct.unpack("<I", val_bytes)[0]
            elif typ == 11 and count == 1: # FLOAT
                interpreted = round(struct.unpack("<f", val_bytes)[0], 6)
            elif typ == 7:
                interpreted = val_bytes.hex()

            entries.append({
                "tag_hex": f"0x{tag:04X}",
                "tag_dec": tag,
                "type": IFD_TYPES.get(typ, str(typ)),
                "count": count,
                "value_bytes_hex": val_bytes.hex(),
                "interpreted_value": interpreted,
            })

        return {
            "num_ifd_entries": num_entries,
            "raw_hex": data.hex(),
            "raw_length_bytes": len(data),
            "entries": entries,
        }
    except Exception as e:
        return {"_error": str(e), "raw_hex": data[:40].hex()}


# ══════════════════════════════════════════════════════════════════════════════
#  MÉTRICAS FOTOGRAMÉTRICAS DERIVADAS
# ══════════════════════════════════════════════════════════════════════════════

def compute_photogrammetric_metrics(xmp: dict, exif: dict, img_w: int, img_h: int):
    """Calcula GSD, footprint e métricas altimétricasderivadas."""

    focal_mm = exif.get("FocalLength")
    if not focal_mm:
        return {"_note": "FocalLength não disponível para cálculo"}

    # Tenta estimar tamanho do sensor a partir do equivalente 35mm
    fl_35 = exif.get("FocalLengthIn35mmFilm")
    if fl_35 and fl_35 > 0 and focal_mm > 0:
        # crop_factor = fl_35 / focal_mm
        # diagonal_35mm = 43.27mm; sensor_diag = 43.27/crop_factor
        crop = fl_35 / focal_mm
        sensor_diag = 43.27 / crop
        # Assume aspect ratio do sensor = imagem
        ar = img_w / img_h
        sensor_h = sensor_diag / math.sqrt(1 + ar ** 2)
        sensor_w = sensor_h * ar
    else:
        sensor_w, sensor_h = None, None

    agl = xmp.get("relative_altitude_m")
    abs_alt = xmp.get("absolute_altitude_m")
    lrf_abs = xmp.get("lrf_target_abs_alt_m")
    lrf_agl = xmp.get("lrf_target_alt_agl_m")
    lrf_dist = xmp.get("lrf_target_distance_m")

    metrics = {
        "sensor_width_mm_estimated": round(sensor_w, 4) if sensor_w else None,
        "sensor_height_mm_estimated": round(sensor_h, 4) if sensor_h else None,
        "pixel_pitch_um_estimated": round(sensor_w / img_w * 1000, 3) if sensor_w else None,
        "agl_from_relative_altitude_m": agl,
        "agl_from_lrf_slant_distance_m": lrf_dist,
        "agl_from_rtk_abs_diff_m": round(abs_alt - lrf_abs, 3) if abs_alt and lrf_abs else None,
        "canopy_height_estimated_m": round(agl - lrf_dist, 3) if agl and lrf_dist else None,
        "ground_elevation_at_lrf_target_wgs84_m": round(lrf_abs - lrf_agl, 3) if lrf_abs and lrf_agl else None,
        "takeoff_ground_elevation_wgs84_m": round(abs_alt - agl, 3) if abs_alt and agl else None,
    }

    if sensor_w and agl and focal_mm:
        pixel_pitch_mm = sensor_w / img_w
        gsd_mm = (agl * 1000 * pixel_pitch_mm) / focal_mm
        gsd_cm = gsd_mm / 10
        fp_w = gsd_cm * img_w / 100
        fp_h = gsd_cm * img_h / 100
        metrics.update({
            "gsd_mm_per_pixel": round(gsd_mm, 4),
            "gsd_cm_per_pixel": round(gsd_cm, 4),
            "footprint_width_m": round(fp_w, 2),
            "footprint_height_m": round(fp_h, 2),
            "footprint_area_m2": round(fp_w * fp_h, 1),
        })

    return metrics

def extract_all(image_path: str) -> dict:
    """Extrai todos os metadados de uma imagem DJI e retorna dict completo."""

    path = Path(image_path)
    if not path.exists():
        raise FileNotFoundError(f"Arquivo não encontrado: {image_path}")

    with open(path, "rb") as f:
        raw_data = f.read()

    # ── Abre imagem ────────────────────────────────────────────────────────
    img = Image.open(path)
    img_w, img_h = img.size

    # ── Extrai camadas ─────────────────────────────────────────────────────
    exif_raw = extract_exif(img)
    gps_parsed = parse_gps_section(exif_raw)
    xmp_raw_str, xmp_attrs = extract_xmp(raw_data)
    dji_xmp = parse_dji_xmp(xmp_attrs)
    jpeg_segments = scan_jpeg_segments(raw_data)

    # MakerNote: recupera bytes brutos do EXIF
    makernote_field = exif_raw.get("MakerNote")
    if isinstance(makernote_field, dict) and "hex" in makernote_field:
        mn_bytes = bytes.fromhex(makernote_field["hex"])
    else:
        mn_bytes = None
    makernote_decoded = parse_makernote(mn_bytes) if mn_bytes else None

    # Métricas derivadas
    derived = compute_photogrammetric_metrics(dji_xmp, exif_raw, img_w, img_h)

    # ── Monta JSON final ───────────────────────────────────────────────────
    output = {
        "_extraction_info": {
            "source_file": path.name,
            "source_path_absolute": str(path.resolve()),
            "file_size_bytes": path.stat().st_size,
            "file_size_mb": round(path.stat().st_size / 1024 / 1024, 4),
            "metadata_layers_extracted": [
                "EXIF IFD0 + SubIFD + GPS IFD",
                "XMP/RDF DJI proprietary namespace",
                "DJI MakerNote IFD (binary little-endian)",
                "JPEG segment map (binary scan)",
            ],
            "total_jpeg_segments_found": len(jpeg_segments),
            "eoi_count_frames": sum(1 for s in jpeg_segments if "EOI" in s["name"]),
            "jpeg_format_detected": (
                "MPO (Multi-Picture Object)"
                if sum(1 for s in jpeg_segments if "EOI" in s["name"]) > 1
                else "JPEG"
            ),
        },
        "image_properties": {
            "width_px": img_w,
            "height_px": img_h,
            "total_pixels": img_w * img_h,
            "megapixels": round(img_w * img_h / 1e6, 2),
            "color_mode": img.mode,
            "format_detected": img.format,
            "color_space_code": exif_raw.get("ColorSpace"),
            "color_space_description": (
                "sRGB" if exif_raw.get("ColorSpace") == 1 else
                "Uncalibrated" if exif_raw.get("ColorSpace") == 0xFFFF else
                str(exif_raw.get("ColorSpace"))
            ),
            "x_resolution_dpi": exif_raw.get("XResolution"),
            "y_resolution_dpi": exif_raw.get("YResolution"),
            "resolution_unit": exif_raw.get("ResolutionUnit"),
            "ycbcr_positioning": exif_raw.get("YCbCrPositioning"),
            "orientation": exif_raw.get("Orientation"),
            "image_description": exif_raw.get("ImageDescription"),
        },
        "device_identification": {
            "manufacturer": exif_raw.get("Make"),
            "model_exif": exif_raw.get("Model"),
            "unique_camera_model": exif_raw.get("UniqueCameraModel"),
            "product_name_xmp": dji_xmp.get("product_name"),
            "drone_model_xmp": dji_xmp.get("drone_model"),
            "drone_serial_number": dji_xmp.get("drone_serial_number") or exif_raw.get("BodySerialNumber"),
            "camera_serial_number": dji_xmp.get("camera_serial_number"),
            "firmware_version": exif_raw.get("Software"),
            "xmp_schema_version": xmp_attrs.get("xmp:ModifyDate", ""),  # fallback
            "dji_xmp_spec_version": dji_xmp.get("dji_xmp_spec_version"),
            "image_source": dji_xmp.get("image_source"),
            "camera_file_type": xmp_attrs.get("Camera:FileType"),
            "shutter_type": dji_xmp.get("shutter_type"),
            "shutter_total_count": dji_xmp.get("shutter_count"),
        },
        "datetime": {
            "datetime_local_exif": exif_raw.get("DateTime"),
            "datetime_original_exif": exif_raw.get("DateTimeOriginal"),
            "datetime_digitized_exif": exif_raw.get("DateTimeDigitized"),
            "xmp_modify_date": xmp_attrs.get("xmp:ModifyDate"),
            "xmp_create_date": xmp_attrs.get("xmp:CreateDate"),
            "utc_at_exposure_precise_xmp": dji_xmp.get("utc_at_exposure"),
        },
        "gps_position_exif": gps_parsed,
        "gps_position_xmp": {
            "source": "XMP DJI namespace (RTK-fused)",
            "gps_status": dji_xmp.get("gps_status"),
            "altitude_type": dji_xmp.get("altitude_type"),
            "rtk_flag": dji_xmp.get("rtk_flag"),
            "rtk_flag_description": (
